/**
 * Set the SelectTool instance for selection visualization
 * @param {SelectTool} tool - The SelectTool instance
 */
setSelectionTool(tool) {
  this.selectionTool = tool;
  this.render();
}

/**
 * Draw selection outline (marching ants) if selection is active
 * @private
 */
_drawSelectionOutline() {
  if (!this.selectionTool) {
    return;
  }

  const bounds = this.selectionTool.getSelectionBounds();
  if (!bounds) {
    return; // No active selection
  }

  // Convert cell bounds to pixel coordinates
  const pixelX = bounds.x * this.cellSizePixels - this.offsetX;
  const pixelY = bounds.y * this.cellSizePixels - this.offsetY;
  const pixelWidth = bounds.width * this.cellSizePixels;
  const pixelHeight = bounds.height * this.cellSizePixels;

  // Draw marching ants outline (dashed line with animation)
  this.ctx.strokeStyle = '#FFFF00'; // Bright yellow
  this.ctx.lineWidth = 1;
  this.ctx.setLineDash([4, 4]); // 4px dash, 4px gap

  // Animate dash offset for marching effect
  const dashOffset = (this._animationFrame % 8) * 0.5;
  this.ctx.lineDashOffset = -dashOffset;

  // Draw the rectangle outline
  this.ctx.strokeRect(pixelX, pixelY, pixelWidth, pixelHeight);

  // Reset line dash
  this.ctx.setLineDash([]);
  this.ctx.lineDashOffset = 0;
}
```

/**
 * Enable pan mode (Space key pressed)
 */
enablePanMode() {
  this.panMode = true;
  // Change cursor to pan cursor
  const canvasElement = this.canvas.getCanvasElement();
  if (canvasElement) {
    canvasElement.style.cursor = 'grab';
  }
}

/**
 * Disable pan mode (Space key released)
 */
disablePanMode() {
  this.panMode = false;
  // Restore normal cursor
  const canvasElement = this.canvas.getCanvasElement();
  if (canvasElement) {
    canvasElement.style.cursor = 'crosshair';
  }
}

/**
 * Start pan drag operation
 * @param {number} x - Starting pixel X coordinate
 * @param {number} y - Starting pixel Y coordinate
 */
startPan(x, y) {
  this.panStartX = x;
  this.panStartY = y;
}

/**
 * Continue pan drag operation
 * @param {number} x - Current pixel X coordinate
 * @param {number} y - Current pixel Y coordinate
 */
pan(x, y) {
  if (!this.panMode) {
    return;
  }

  const deltaX = x - this.panStartX;
  const deltaY = y - this.panStartY;

  // Update canvas offset (pan moves view, so offset moves opposite direction)
  const newOffsetX = this.canvas.offsetX - deltaX;
  const newOffsetY = this.canvas.offsetY - deltaY;

  this.canvas.setOffset(newOffsetX, newOffsetY);

  // Update pan start for next drag iteration
  this.panStartX = x;
  this.panStartY = y;
}

/**
 * End pan drag operation
 */
endPan() {
  // no-op
}
```

```javascript
/**
 * Set the LayerStack for multi-layer composition
 * When set, Canvas will composite layers instead of using single cell map
 * @param {LayerStack} layerStack - The LayerStack instance
 */
setLayerStack(layerStack) {
  this.layerStack = layerStack;
  this.useLayerStack = true;
  this.render();
}

/**
 * Get cell data, compositing from LayerStack if available
 * Returns topmost visible layer's cell, or transparent if no visible layer has content
 * @param {number} x - Cell X coordinate
 * @param {number} y - Cell Y coordinate
 * @returns {Object} Cell data {glyph, fg, bg}
 */
getCell(x, y) {
  // If using LayerStack, composite from visible layers
  if (this.useLayerStack && this.layerStack) {
    // Iterate layers from top to bottom, return first visible layer with content
    const layers = this.layerStack.layers;
    for (let i = layers.length - 1; i >= 0; i--) {
      const layer = layers[i];
      if (!layer.visible) {
        continue; // Skip hidden layers
      }
      const cell = layer.getCell(x, y);
      if (cell && cell.glyph !== 0) {
        return cell; // Return cell from topmost layer with non-transparent glyph
      }
    }
    // If no visible layer has content, return transparent
    return {
      glyph: 0,
      fg: [255, 255, 255],
      bg: [0, 0, 0],
    };
  }

  // Original single-layer behavior
  this._validateCoordinates(x, y);
  const key = `${x},${y}`;
  const stored = this.cells.get(key) || {
    glyph: 0,
    fg: [255, 255, 255],
    bg: [0, 0, 0],
  };

  return {
    glyph: stored.glyph,
    fg: [...stored.fg],
    bg: [...stored.bg],
  };
}

/**
 * Set cell data, applies to active layer if using LayerStack
 * @param {number} x - Cell X coordinate
 * @param {number} y - Cell Y coordinate
 * @param {number} glyph - CP437 glyph code
 * @param {Array<number>} fg - Foreground color [R, G, B]
 * @param {Array<number>} bg - Background color [R, G, B]
 */
setCell(x, y, glyph, fg, bg) {
  // If using LayerStack, apply to active layer
  if (this.useLayerStack && this.layerStack) {
    const activeLayer = this.layerStack.getActiveLayer();
    if (activeLayer) {
      activeLayer.setCell(x, y, glyph, fg, bg);
    }
    return;
  }

  // Original single-layer behavior
  this._validateCoordinates(x, y);
  const key = `${x},${y}`;
  this.cells.set(key, {
    glyph: glyph & 0xFF,
    fg: [...fg],
    bg: [...bg],
  });
}
```
